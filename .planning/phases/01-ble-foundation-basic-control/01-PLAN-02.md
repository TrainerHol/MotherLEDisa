**DeviceViewModel.kt** - State management for device list:
```kotlin
package com.motherledisa.ui.device

import android.text.format.DateUtils
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.motherledisa.data.ble.DeviceScanner
import com.motherledisa.data.ble.TowerConnectionManager
import com.motherledisa.data.local.TowerConfigEntity
import com.motherledisa.data.repository.TowerRepository
import com.motherledisa.domain.model.ConnectionState
import com.motherledisa.domain.model.DiscoveredDevice
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

@HiltViewModel
class DeviceViewModel @Inject constructor(
    private val deviceScanner: DeviceScanner,
    private val connectionManager: TowerConnectionManager,
    private val towerRepository: TowerRepository
) : ViewModel() {

    private val _isScanning = MutableStateFlow(false)
    private val _permissionsGranted = MutableStateFlow(false)
    private val _showPermissionDialog = MutableStateFlow(false)
    private val _renameDialogDevice = MutableStateFlow<DeviceListItem?>(null)

    // Merge scanned devices with known devices (D-05)
    private val mergedDevices: StateFlow<List<DeviceListItem>> = combine(
        deviceScanner.discoveredDevices,
        towerRepository.knownDevices,
        connectionManager.connectedTowers
    ) { scanned, known, connected ->
        val connectedAddresses = connected.map { it.address }.toSet()
        val scannedAddresses = scanned.map { it.address }.toSet()

        // Convert scanned devices
        val availableDevices = scanned.map { device ->
            val knownConfig = known.find { it.address == device.address }
            DeviceListItem.Available(
                address = device.address,
                displayName = knownConfig?.customName ?: device.name,
                rssi = device.rssi,
                isConnected = device.address in connectedAddresses,
                isKnown = knownConfig != null
            )
        }

        // Add known devices not currently visible
        val knownNotVisible = known
            .filterNot { it.address in scannedAddresses }
            .map { config ->
                DeviceListItem.Known(
                    address = config.address,
                    displayName = config.customName ?: config.originalName,
                    lastConnectedFormatted = formatRelativeTime(config.lastConnected),
                    isConnected = config.address in connectedAddresses
                )
            }

        // Merge: available devices first, then known not visible
        (availableDevices + knownNotVisible).sortedByDescending { item ->
            when (item) {
                is DeviceListItem.Available -> if (item.isConnected) Long.MAX_VALUE else item.rssi?.toLong() ?: 0L
                is DeviceListItem.Known -> 0L  // Sort to bottom
            }
        }
    }.stateIn(viewModelScope, SharingStarted.Lazily, emptyList())

    val uiState: StateFlow<DeviceListUiState> = combine(
        mergedDevices,
        _isScanning,
        connectionManager.connectionState,
        _permissionsGranted,
        _showPermissionDialog,
        _renameDialogDevice
    ) { devices, scanning, connState, permsGranted, showDialog, renameDevice ->
        DeviceListUiState(
            devices = devices,
            isScanning = scanning,
            connectionState = connState,
            permissionsGranted = permsGranted,
            showPermissionDialog = showDialog,
            renameDialogDevice = renameDevice
        )
    }.stateIn(viewModelScope, SharingStarted.Eagerly, DeviceListUiState())

    fun startScanning() {
        if (!_permissionsGranted.value) {
            _showPermissionDialog.value = true
            return
        }
        _isScanning.value = true
        deviceScanner.startContinuousScan()
    }

    fun stopScanning() {
        _isScanning.value = false
        deviceScanner.stopScan()
    }

    fun connect(address: String) {
        viewModelScope.launch {
            connectionManager.connect(address)
        }
    }

    fun disconnect(address: String) {
        val tower = connectionManager.connectedTowers.value.find { it.address == address }
        if (tower != null) {
            viewModelScope.launch {
                connectionManager.disconnect(tower)
            }
        }
    }

    fun onPermissionsResult(granted: Boolean) {
        _permissionsGranted.value = granted
        _showPermissionDialog.value = false
        if (granted) {
            startScanning()
        }
    }

    fun dismissPermissionDialog() {
        _showPermissionDialog.value = false
    }

    fun showRenameDialog(device: DeviceListItem) {
        _renameDialogDevice.value = device
    }

    fun dismissRenameDialog() {
        _renameDialogDevice.value = null
    }

    fun renameDevice(address: String, newName: String) {
        viewModelScope.launch {
            towerRepository.renameDevice(address, newName)
            _renameDialogDevice.value = null
        }
    }

    private fun formatRelativeTime(timestamp: Long): String {
        return DateUtils.getRelativeTimeSpanString(
            timestamp,
            System.currentTimeMillis(),
            DateUtils.MINUTE_IN_MILLIS
        ).toString()
    }
}
```

**SignalStrengthBars.kt** - RSSI visualization per UI-SPEC:
```kotlin
package com.motherledisa.ui.components

import androidx.compose.foundation.Canvas
import androidx.compose.foundation.layout.size
import androidx.compose.material3.MaterialTheme
import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import androidx.compose.ui.geometry.CornerRadius
import androidx.compose.ui.geometry.Offset
import androidx.compose.ui.geometry.Size
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.semantics.contentDescription
import androidx.compose.ui.semantics.semantics
import androidx.compose.ui.unit.dp

@Composable
fun SignalStrengthBars(
    rssi: Int?,
    modifier: Modifier = Modifier
) {
    // RSSI to bars mapping from UI-SPEC
    val filledBars = when {
        rssi == null -> 0
        rssi >= -50 -> 4  // Excellent
        rssi >= -65 -> 3  // Good
        rssi >= -80 -> 2  // Fair
        else -> 1         // Weak
    }

    val strengthLabel = when (filledBars) {
        4 -> "excellent"
        3 -> "good"
        2 -> "fair"
        1 -> "weak"
        else -> "unknown"
    }

    val filledColor = MaterialTheme.colorScheme.onSurface
    val unfilledColor = filledColor.copy(alpha = 0.38f)

    Canvas(
        modifier = modifier
            .size(width = 24.dp, height = 20.dp)
            .semantics { contentDescription = "Signal strength $strengthLabel" }
    ) {
        val barWidth = 4.dp.toPx()
        val barGap = 2.dp.toPx()
        val totalWidth = 4 * barWidth + 3 * barGap
        val startX = (size.width - totalWidth) / 2

        // Bar heights: 25%, 50%, 75%, 100% of available height
        val heights = listOf(0.25f, 0.5f, 0.75f, 1.0f)

        heights.forEachIndexed { index, heightPercent ->
            val barHeight = size.height * heightPercent
            val x = startX + index * (barWidth + barGap)
            val y = size.height - barHeight
            val isFilled = index < filledBars

            drawRoundRect(
                color = if (isFilled) filledColor else unfilledColor,
                topLeft = Offset(x, y),
                size = Size(barWidth, barHeight),
                cornerRadius = CornerRadius(2.dp.toPx())
            )
        }
    }
}
```

```kotlin
package com.motherledisa.ui.device

import android.Manifest
import android.os.Build
import androidx.activity.compose.rememberLauncherForActivityResult
import androidx.activity.result.contract.ActivityResultContracts
import androidx.compose.foundation.clickable
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.foundation.lazy.items
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.platform.LocalLifecycleOwner
import androidx.compose.ui.text.style.TextAlign
import androidx.compose.ui.unit.dp
import androidx.hilt.navigation.compose.hiltViewModel
import androidx.lifecycle.Lifecycle
import androidx.lifecycle.LifecycleEventObserver
import com.motherledisa.domain.model.ConnectionState
import com.motherledisa.ui.components.PermissionDialog
import com.motherledisa.ui.components.RenameDialog

@Composable
fun DeviceListScreen(
    viewModel: DeviceViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsState()
    val lifecycleOwner = LocalLifecycleOwner.current

    // Permission request launcher
    val permissionLauncher = rememberLauncherForActivityResult(
        ActivityResultContracts.RequestMultiplePermissions()
    ) { permissions ->
        val allGranted = permissions.values.all { it }
        viewModel.onPermissionsResult(allGranted)
    }

    // BLE permissions based on Android version
    val blePermissions = remember {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
            arrayOf(
                Manifest.permission.BLUETOOTH_SCAN,
                Manifest.permission.BLUETOOTH_CONNECT,
                Manifest.permission.ACCESS_FINE_LOCATION
            )
        } else {
            arrayOf(
                Manifest.permission.BLUETOOTH,
                Manifest.permission.BLUETOOTH_ADMIN,
                Manifest.permission.ACCESS_FINE_LOCATION
            )
        }
    }

    // Start/stop scanning with screen lifecycle (D-01)
    DisposableEffect(lifecycleOwner) {
        val observer = LifecycleEventObserver { _, event ->
            when (event) {
                Lifecycle.Event.ON_RESUME -> viewModel.startScanning()
                Lifecycle.Event.ON_PAUSE -> viewModel.stopScanning()
                else -> {}
            }
        }
        lifecycleOwner.lifecycle.addObserver(observer)
        onDispose {
            lifecycleOwner.lifecycle.removeObserver(observer)
        }
    }

    // Permission dialog (D-17)
    if (uiState.showPermissionDialog) {
        PermissionDialog(
            onGrant = { permissionLauncher.launch(blePermissions) },
            onDismiss = { viewModel.dismissPermissionDialog() }
        )
    }

    // Rename dialog (D-16)
    uiState.renameDialogDevice?.let { device ->
        RenameDialog(
            currentName = device.displayName,
            onRename = { newName -> viewModel.renameDevice(device.address, newName) },
            onDismiss = { viewModel.dismissRenameDialog() }
        )
    }

    // Main content
    Column(modifier = Modifier.fillMaxSize()) {
        // Scanning indicator
        if (uiState.isScanning) {
            LinearProgressIndicator(modifier = Modifier.fillMaxWidth())
        }

        if (uiState.devices.isEmpty()) {
            // Empty state (D-04)
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .clickable { viewModel.startScanning() },
                contentAlignment = Alignment.Center
            ) {
                Text(
                    text = "No devices found. Tap to scan again.",
                    style = MaterialTheme.typography.bodyLarge,
                    textAlign = TextAlign.Center,
                    modifier = Modifier.padding(32.dp)
                )
            }
        } else {
            // Device list
            LazyColumn(
                modifier = Modifier.fillMaxSize()
            ) {
                items(
                    items = uiState.devices,
                    key = { it.address }
                ) { device ->
                    val isConnecting = uiState.connectionState is ConnectionState.Connecting &&
                        (uiState.connectionState as ConnectionState.Connecting).address == device.address

                    DeviceListItemRow(
                        device = device,
                        isConnecting = isConnecting,
                        onClick = {
                            if (device.isConnected) {
                                viewModel.disconnect(device.address)
                            } else {
                                viewModel.connect(device.address)
                            }
                        },
                        onLongClick = { viewModel.showRenameDialog(device) }
                    )
                    HorizontalDivider()
                }
            }
        }

        // Connection error snackbar
        val connectionState = uiState.connectionState
        if (connectionState is ConnectionState.Error) {
            Snackbar(
                modifier = Modifier.padding(16.dp)
            ) {
                Text("Connection failed: ${connectionState.message}")
            }
        }
    }
}
```
