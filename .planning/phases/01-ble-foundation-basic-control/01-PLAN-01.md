**MelkProtocol.kt** - ELK-BLEDOM protocol command builder:
```kotlin
object MelkProtocol {
    const val SERVICE_UUID = "0000fff0-0000-1000-8000-00805f9b34fb"
    const val CHAR_UUID = "0000fff3-0000-1000-8000-00805f9b34fb"

    private val COMPATIBLE_PREFIXES = listOf("ELK-", "MELK-", "LEDBLE", "ELK-BULB", "ELK-LAMPL")

    fun isCompatibleDevice(name: String?): Boolean =
        COMPATIBLE_PREFIXES.any { name?.startsWith(it, ignoreCase = true) == true }

    fun isMelkDevice(name: String?): Boolean =
        name?.startsWith("MELK-", ignoreCase = true) == true

    // Init sequence for MELK devices (required before control commands work)
    val INIT_COMMAND_1 = byteArrayOf(0x7e, 0x07, 0x83.toByte())
    val INIT_COMMAND_2 = byteArrayOf(0x7e, 0x04, 0x04)

    fun powerOn() = byteArrayOf(0x7e, 0x04, 0x04, 0xf0.toByte(), 0x00, 0x01, 0xff.toByte(), 0x00, 0xef.toByte())
    fun powerOff() = byteArrayOf(0x7e, 0x04, 0x04, 0x00, 0x00, 0x00, 0xff.toByte(), 0x00, 0xef.toByte())

    fun setColor(r: Int, g: Int, b: Int) = byteArrayOf(
        0x7e, 0x07, 0x05, 0x03,
        r.coerceIn(0, 255).toByte(),
        g.coerceIn(0, 255).toByte(),
        b.coerceIn(0, 255).toByte(),
        0x10, 0xef.toByte()
    )

    fun setBrightness(level: Int) = byteArrayOf(
        0x7e, 0x00, 0x01,
        level.coerceIn(0, 100).toByte(),
        0x00, 0x00, 0x00, 0x00, 0xef.toByte()
    )

    fun setEffect(effectId: Int, speed: Int = 50) = byteArrayOf(
        0x7e, 0x00, 0x03,
        effectId.coerceIn(0, 255).toByte(),
        speed.coerceIn(0, 100).toByte(),
        0x00, 0x00, 0x00, 0xef.toByte()
    )
}
```

sealed class BleCommand(val data: ByteArray) {
    class PowerOn : BleCommand(MelkProtocol.powerOn())

class TowerRepository @Inject constructor(
    private val towerConfigDao: TowerConfigDao
) {
    val knownDevices: Flow<List<TowerConfigEntity>> = towerConfigDao.getAllKnownDevices()

    suspend fun saveConnectedDevice(address: String, name: String) {
        val existing = towerConfigDao.getByAddress(address)
        val config = TowerConfigEntity(
            address = address,
            customName = existing?.customName,
            originalName = name,
            lastConnected = System.currentTimeMillis(),
            position = existing?.position ?: 0
        )
        towerConfigDao.upsert(config)
    }

    suspend fun updateLastConnected(address: String) {
        towerConfigDao.updateLastConnected(address, System.currentTimeMillis())
    }

    suspend fun renameDevice(address: String, customName: String) {
        towerConfigDao.updateCustomName(address, customName)
    }

    suspend fun getConfig(address: String): TowerConfigEntity? {
        return towerConfigDao.getByAddress(address)
    }
}
