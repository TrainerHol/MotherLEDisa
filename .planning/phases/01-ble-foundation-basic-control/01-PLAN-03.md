```kotlin
package com.motherledisa.domain.usecase

import com.motherledisa.data.ble.BleCommand
import com.motherledisa.data.ble.TowerConnectionManager
import com.motherledisa.domain.model.Tower
import javax.inject.Inject

class TogglePowerUseCase @Inject constructor(
    private val connectionManager: TowerConnectionManager
) {
    suspend operator fun invoke(tower: Tower, turnOn: Boolean) {
        val command = if (turnOn) BleCommand.PowerOn() else BleCommand.PowerOff()
        tower.commandQueue.enqueue(command)
    }

    suspend fun invokeAll(turnOn: Boolean) {
        connectionManager.connectedTowers.value.forEach { tower ->
            invoke(tower, turnOn)
        }
    }
}
```

```kotlin
package com.motherledisa.ui.control

import androidx.compose.ui.graphics.Color
import androidx.compose.ui.graphics.toArgb
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.motherledisa.data.ble.TowerConnectionManager
import com.motherledisa.domain.model.EffectLibrary
import com.motherledisa.domain.model.Tower
import com.motherledisa.domain.usecase.SetBrightnessUseCase
import com.motherledisa.domain.usecase.SetColorUseCase
import com.motherledisa.domain.usecase.SetEffectUseCase
import com.motherledisa.domain.usecase.TogglePowerUseCase
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.FlowPreview
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

@OptIn(FlowPreview::class)
@HiltViewModel
class ControlViewModel @Inject constructor(
    private val connectionManager: TowerConnectionManager,
    private val togglePowerUseCase: TogglePowerUseCase,
    private val setColorUseCase: SetColorUseCase,
    private val setBrightnessUseCase: SetBrightnessUseCase,
    private val setEffectUseCase: SetEffectUseCase
) : ViewModel() {

    private val _selectedDevice = MutableStateFlow<SelectedDevice?>(null)

    // Local state for immediate UI feedback
    private val _isPoweredOn = MutableStateFlow(true)
    private val _currentColor = MutableStateFlow(Color.White)
    private val _brightness = MutableStateFlow(100)
    private val _activeEffectId = MutableStateFlow<Int?>(null)
    private val _effectSpeed = MutableStateFlow(50)

    // Derive tower state for preview
    private val towerState: StateFlow<TowerState?> = combine(
        _selectedDevice,
        _isPoweredOn,
        _currentColor,
        _brightness,
        _activeEffectId,
        _effectSpeed
    ) { selected, powered, color, brightness, effectId, speed ->
        when (selected) {
            is SelectedDevice.Single -> TowerState(
                address = selected.tower.address,
                name = selected.tower.name,
                isPoweredOn = powered,
                currentColor = color,
                brightness = brightness,
                activeEffectId = effectId,
                effectSpeed = speed
            )
            is SelectedDevice.All -> TowerState(
                address = "all",
                name = "All Devices",
                isPoweredOn = powered,
                currentColor = color,
                brightness = brightness,
                activeEffectId = effectId,
                effectSpeed = speed
            )
            null -> null
        }
    }.stateIn(viewModelScope, SharingStarted.Lazily, null)

    val uiState: StateFlow<ControlUiState> = combine(
        connectionManager.connectedTowers,
        _selectedDevice,
        towerState
    ) { towers, selected, state ->
        // Auto-select first device if none selected
        val actualSelected = selected ?: towers.firstOrNull()?.let { SelectedDevice.Single(it) }
        if (selected == null && actualSelected != null) {
            _selectedDevice.value = actualSelected
        }

        ControlUiState(
            connectedDevices = towers,
            selectedDevice = actualSelected,
            towerState = state,
            effectsByCategory = EffectLibrary.BY_CATEGORY,
            hasMultipleDevices = towers.size > 1
        )
    }.stateIn(viewModelScope, SharingStarted.Eagerly, ControlUiState())

    init {
        // Debounce brightness changes to ~30fps (D-10)
        _brightness
            .debounce(33)  // ~30fps
            .distinctUntilChanged()
            .onEach { brightness ->
                sendBrightnessCommand(brightness)
            }
            .launchIn(viewModelScope)

        // Debounce color changes similarly
        _currentColor
            .debounce(33)
            .distinctUntilChanged()
            .onEach { color ->
                sendColorCommand(color)
            }
            .launchIn(viewModelScope)

        // Debounce effect speed
        _effectSpeed
            .debounce(33)
            .distinctUntilChanged()
            .onEach { speed ->
                val effectId = _activeEffectId.value
                if (effectId != null) {
                    sendEffectCommand(effectId, speed)
                }
            }
            .launchIn(viewModelScope)
    }

    fun selectDevice(device: SelectedDevice) {
        _selectedDevice.value = device
    }

    fun togglePower() {
        val newState = !_isPoweredOn.value
        _isPoweredOn.value = newState

        viewModelScope.launch {
            when (val selected = _selectedDevice.value) {
                is SelectedDevice.Single -> togglePowerUseCase(selected.tower, newState)
                is SelectedDevice.All -> togglePowerUseCase.invokeAll(newState)
                null -> {}
            }
        }
    }

    fun setColor(color: Color) {
        _currentColor.value = color
        // Debounced send handled by init block
    }

    fun setBrightness(level: Int) {
        _brightness.value = level
        // Debounced send handled by init block
    }

    fun setEffect(effectId: Int) {
        val previousEffect = _activeEffectId.value
        if (previousEffect == effectId) {
            // Toggle off if same effect tapped
            _activeEffectId.value = null
        } else {
            _activeEffectId.value = effectId
            viewModelScope.launch {
                sendEffectCommand(effectId, _effectSpeed.value)
            }
        }
    }

    fun setEffectSpeed(speed: Int) {
        _effectSpeed.value = speed
        // Debounced send handled by init block
    }

    private suspend fun sendColorCommand(color: Color) {
        val argb = color.toArgb()
        val r = (argb shr 16) and 0xFF
        val g = (argb shr 8) and 0xFF
        val b = argb and 0xFF

        when (val selected = _selectedDevice.value) {
            is SelectedDevice.Single -> setColorUseCase(selected.tower, r, g, b)
            is SelectedDevice.All -> setColorUseCase.invokeAll(r, g, b)
            null -> {}
        }
    }

    private suspend fun sendBrightnessCommand(level: Int) {
        when (val selected = _selectedDevice.value) {
            is SelectedDevice.Single -> setBrightnessUseCase(selected.tower, level)
            is SelectedDevice.All -> setBrightnessUseCase.invokeAll(level)
            null -> {}
        }
    }

    private suspend fun sendEffectCommand(effectId: Int, speed: Int) {
        when (val selected = _selectedDevice.value) {
            is SelectedDevice.Single -> setEffectUseCase(selected.tower, effectId, speed)
            is SelectedDevice.All -> setEffectUseCase.invokeAll(effectId, speed)
            null -> {}
        }
    }
}
```

```kotlin
package com.motherledisa.ui.control

import androidx.compose.animation.animateColorAsState
import androidx.compose.animation.core.tween
import androidx.compose.foundation.Canvas
import androidx.compose.foundation.layout.fillMaxWidth
import androidx.compose.foundation.layout.height
import androidx.compose.runtime.Composable
import androidx.compose.runtime.getValue
import androidx.compose.ui.Modifier
import androidx.compose.ui.geometry.CornerRadius
import androidx.compose.ui.geometry.Offset
import androidx.compose.ui.geometry.Size
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.graphics.drawscope.Stroke
import androidx.compose.ui.unit.dp

@Composable
fun TowerPreviewCanvas(
    towerState: TowerState,
    modifier: Modifier = Modifier
) {
    // Animate color transitions (300ms per UI-SPEC)
    val animatedColor by animateColorAsState(
        targetValue = if (towerState.isPoweredOn) towerState.currentColor else Color.DarkGray,
        animationSpec = tween(300),
        label = "color"
    )

    Canvas(
        modifier = modifier
            .fillMaxWidth()
            .height(300.dp)
    ) {
        val segmentCount = towerState.segmentCount
        val segmentHeight = size.height / (segmentCount + 0.5f)  // Extra space for base
        val segmentWidth = size.width * 0.5f
        val offsetX = (size.width - segmentWidth) / 2
        val cornerRadius = 8.dp.toPx()
        val gap = 4.dp.toPx()

        // Draw tower segments from bottom to top
        for (i in 0 until segmentCount) {
            val segmentColor = if (towerState.isPoweredOn) {
                towerState.segmentColors.getOrNull(i) ?: animatedColor
            } else {
                Color.DarkGray
            }

            // Apply brightness to color
            val brightnessMultiplier = towerState.brightness / 100f
            val adjustedColor = segmentColor.copy(
                red = segmentColor.red * brightnessMultiplier,
                green = segmentColor.green * brightnessMultiplier,
                blue = segmentColor.blue * brightnessMultiplier
            )

            val y = size.height - (i + 1.5f) * segmentHeight  // Offset for base

            // Segment fill
            drawRoundRect(
                color = adjustedColor,
                topLeft = Offset(offsetX, y + gap / 2),
                size = Size(segmentWidth, segmentHeight - gap),
                cornerRadius = CornerRadius(cornerRadius, cornerRadius)
            )

            // Subtle border
            drawRoundRect(
                color = Color.White.copy(alpha = 0.2f),
                topLeft = Offset(offsetX, y + gap / 2),
                size = Size(segmentWidth, segmentHeight - gap),
                cornerRadius = CornerRadius(cornerRadius, cornerRadius),
                style = Stroke(width = 1.dp.toPx())
            )
        }

        // Tower base
        drawCircle(
            color = Color(0xFF333333),
            radius = segmentWidth / 2 + 15.dp.toPx(),
            center = Offset(size.width / 2, size.height - 20.dp.toPx())
        )
    }
}
```

```kotlin
import com.motherledisa.ui.theme.TextDisabled

@Composable
fun PowerToggleButton(
    isOn: Boolean,
    onClick: () -> Unit,
    modifier: Modifier = Modifier
) {
    val backgroundColor by animateColorAsState(
        targetValue = if (isOn) SecondaryAccent else TextDisabled,
        animationSpec = tween(200),
        label = "bgColor"
    )

    val contentColor = if (isOn) Color.Black else Color.White

    // Subtle pulse when ON
    val infiniteTransition = rememberInfiniteTransition(label = "pulse")
    val pulseAlpha by infiniteTransition.animateFloat(
        initialValue = 0.6f,
        targetValue = 1f,
        animationSpec = infiniteRepeatable(
            animation = tween(1000, easing = EaseInOutSine),
            repeatMode = RepeatMode.Reverse
        ),
        label = "pulseAlpha"
    )

    Column(
        horizontalAlignment = Alignment.CenterHorizontally,
        modifier = modifier.padding(16.dp)
    ) {
        Box(
            contentAlignment = Alignment.Center,
            modifier = Modifier
                .size(72.dp)
                .shadow(
                    elevation = if (isOn) 8.dp else 2.dp,
                    shape = CircleShape,
                    spotColor = if (isOn) SecondaryAccent.copy(alpha = pulseAlpha) else Color.Transparent
                )
                .clip(CircleShape)
                .background(backgroundColor)
                .border(
                    width = 2.dp,
                    color = if (isOn) SecondaryAccent else Color.Transparent,
                    shape = CircleShape
                )
                .clickable(onClick = onClick)
                .semantics { contentDescription = "Power ${if (isOn) "on" else "off"}" }
        ) {
            Icon(
                imageVector = Icons.Default.Power,
                contentDescription = null,
                tint = contentColor,
                modifier = Modifier.size(36.dp)
            )
        }

        Text(
            text = if (isOn) "ON" else "OFF",
            style = MaterialTheme.typography.labelLarge,
            color = MaterialTheme.colorScheme.onSurface,
            modifier = Modifier.padding(top = 8.dp)
        )
    }
}
```

```kotlin
@Composable
private fun ColorSwatch(
    color: Color,
    isSelected: Boolean,
    onClick: () -> Unit
) {

        Color(0xFFFF0000) -> "Red"
        Color(0xFFFF7F00) -> "Orange"
        Color(0xFFFFFF00) -> "Yellow"
        Color(0xFF00FF00) -> "Green"
        Color(0xFF00FFFF) -> "Cyan"
        Color(0xFF0000FF) -> "Blue"
        Color(0xFF8B00FF) -> "Purple"
        Color(0xFFFFFFFF) -> "White"
        else -> "Color"
    }

    Box(
        modifier = Modifier
            .size(48.dp)
            .clip(CircleShape)
            .background(color)
            .border(
                width = if (isSelected) 3.dp else 1.dp,
                color = if (isSelected) PrimaryAccent else Color.Gray,
                shape = CircleShape
            )
            .clickable(onClick = onClick)
            .semantics { contentDescription = colorName }
```

```kotlin
    ) {
        OutlinedTextField(
            value = displayText,
            onValueChange = {},
            readOnly = true,

            modifier = Modifier
                .fillMaxWidth()
                .menuAnchor()
        )

        ExposedDropdownMenu(
            expanded = expanded,
            onDismissRequest = { expanded = false }
        ) {
            // "All devices" option
            DropdownMenuItem(
                text = { Text("All devices") },
                onClick = {
                    onDeviceSelected(SelectedDevice.All)
                    expanded = false
                }
            )

                    text = { Text(tower.name) },

                        expanded = false
```

```kotlin
package com.motherledisa.ui.control

import androidx.compose.foundation.layout.*
import androidx.compose.foundation.rememberScrollState
import androidx.compose.foundation.verticalScroll
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.compose.runtime.collectAsState
import androidx.compose.runtime.getValue
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.style.TextAlign
import androidx.compose.ui.unit.dp
import androidx.hilt.navigation.compose.hiltViewModel

@Composable
fun ControlScreen(
    viewModel: ControlViewModel = hiltViewModel()
) {
    val uiState by viewModel.uiState.collectAsState()

    if (uiState.connectedDevices.isEmpty()) {
        // No devices connected
        Box(
            modifier = Modifier.fillMaxSize(),
            contentAlignment = Alignment.Center
        ) {
            Text(
                text = "No devices connected.\nGo to Devices tab to connect.",
                style = MaterialTheme.typography.bodyLarge,
                textAlign = TextAlign.Center,
                color = MaterialTheme.colorScheme.onSurfaceVariant
            )
        }
        return
    }

    Column(
        modifier = Modifier
            .fillMaxSize()
            .verticalScroll(rememberScrollState())
    ) {
        // D-12: Device picker (only when multiple devices)
        if (uiState.hasMultipleDevices) {
            DevicePicker(
                connectedDevices = uiState.connectedDevices,
                selectedDevice = uiState.selectedDevice,
                onDeviceSelected = { viewModel.selectDevice(it) }
            )
        }

        // Tower state for components
        val towerState = uiState.towerState

        if (towerState != null) {
            // D-07: Tower preview at top
            TowerPreviewCanvas(
                towerState = towerState,
                modifier = Modifier.padding(16.dp)
            )

            // D-09: Power toggle
            PowerToggleButton(
                isOn = towerState.isPoweredOn,
                onClick = { viewModel.togglePower() },
                modifier = Modifier.align(Alignment.CenterHorizontally)
            )

            HorizontalDivider(modifier = Modifier.padding(vertical = 8.dp))

            // D-08, D-11: Color picker
            ColorPickerSection(
                selectedColor = towerState.currentColor,
                onColorSelected = { viewModel.setColor(it) }
            )

            HorizontalDivider(modifier = Modifier.padding(vertical = 8.dp))

            // D-10: Brightness slider
            BrightnessSlider(
                brightness = towerState.brightness,
                onBrightnessChanged = { viewModel.setBrightness(it) }
            )

            HorizontalDivider(modifier = Modifier.padding(vertical = 8.dp))

            // D-18, D-19, D-22: Effects section
            EffectsSection(
                effectsByCategory = uiState.effectsByCategory,
                activeEffectId = towerState.activeEffectId,
                onEffectSelected = { viewModel.setEffect(it) }
            )

            // D-20: Speed slider (only when effect active)
            if (towerState.activeEffectId != null) {
                HorizontalDivider(modifier = Modifier.padding(vertical = 8.dp))

                SpeedSlider(
                    speed = towerState.effectSpeed,
                    onSpeedChanged = { viewModel.setEffectSpeed(it) }
                )
            }

            // Bottom padding for navigation bar
            Spacer(modifier = Modifier.height(16.dp))
        }
    }
}
```
